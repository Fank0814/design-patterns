const addDecorator = (fn, before, after) => {
  let isFn = fn => typeof fn === "function";

  if (!isFn(fn)) {
    return () => {};
  }

  return (...args) => {
    let result;
    // æŒ‰ç…§é¡ºåºæ‰§è¡Œâ€œè£…é¥°å‡½æ•°â€

    isFn(after) && after(...args);
    // æœ€åè¿”å›ç»“æœ
    return result;
  };
};
